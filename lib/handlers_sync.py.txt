# Telegram API
def send_message(token, chat_id, text, reply_markup=None):
    """Отправка сообщения через Telegram API"""
    url = f"https://api.telegram.org/bot{token}/sendMessage"
    data = {
        "chat_id": chat_id,
        "text": text,
        "parse_mode": "HTML"
    }
    if reply_markup:
        data["reply_markup"] = json.dumps(reply_markup)
    
    resp = requests.post(url, json=data, timeout=10)
    return resp.json()


def answer_callback(token, callback_query_id, text=None):
    """Ответ на callback query"""
    url = f"https://api.telegram.org/bot{token}/answerCallbackQuery"
    data = {
        "callback_query_id": callback_query_id
    }
    if text:
        data["text"] = text
    if text is None:
        data["show_alert"] = False
    
    resp = requests.post(url, json=data, timeout=10)
    return resp.json()

def handle_update_sync(update_data, token):
    """Главный обработчик обновлений"""
    update = update_data
    
    # Проверка типа обновления
    if "message" in update:
        message = update["message"]
        chat_id = message["chat"]["id"]
        text = message.get("text", "")
        
        # Команда /start
        if text == "/start":
            handle_start_sync(token, chat_id, message["from"].get("first_name", "Пользователь"))
    
    elif "callback_query" in update:
        callback = update["callback_query"]
        chat_id = callback["message"]["chat"]["id"]
        callback_id = callback["id"]
        data = callback.get("data", "")
        
        # Ответ на callback (обязательно!)
        answer_callback(token, callback_id)
        
        # Обработка callback data
        handle_callback_sync(token, chat_id, callback_id, data, callback["from"].get("first_name", "Пользователь"))
